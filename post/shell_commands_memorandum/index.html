# ----------
# 基本
# ----------
% ls
% grep word    # "word"の文字列検索
% wc -l        # =word counts: 単語数

% ls | grep conf | wc -l
17


# ----------
# 移動
# ----------
% cd
% pwd
~/sandbox/prodctions/dev/user
% cd dev prod   # 特定のディレクトリ構造であればジャンプできる
~/sandbox/prodctions/prod/user

# Directory Hash (Zsh特有)
# 移動先ルートパスをメモ書きできる
% hash -d api=/tmp/src/main/go/api
% cd ~api && pwd
~/tmp/src/main/go/api


# ----------------------------------------
# 操作系
# ----------------------------------------
# カーソル移動 (Emacs key風は macOS 特有)
#
#       ^p
#   ^b  |  ^f
#       ^n
#
#   ^d :delete    with Ctrl + d
#   ^h :backspace with Ctrl + h
#   ^a :行頭
#   ^e :行末
#   ^k :カーソルより後ろ（右方向）削除
#   ^w :カーソルより前（左方向）の単語消す


# ----------------------------------------
# 履歴
# ----------------------------------------
% history

# 過去のgit コマンド列挙して、ナンバー実行
% history | grep git
% !{number}


# ----------------------------------------
# 展開
# ----------------------------------------
# - Brace Expansion (zsh特有) :{,}で総当り展開
# - Glob  Expansion (zsh特集) :Tabで確認展開（^xu :で戻せる）
% echo {shop_a,shop_b}.csv
shop_a.csv shop_b.csv

% diff -u {shop_a,shop_b}.csv

% cp -p httpd.conf htppdd.conf.buckup
% cp -p htppd.conf{,.buckup}
