[GSS] 複数対象の文字置換

カラムの表記揺れを直している時、不要文字列を除外したい時に使える。 例えば、

old acc
mo ji -> oji
o mji -> oji

1列目、1~2行いずれも「m」「space」を除外したい。 substituto() や replace() を連続使用すると割とゴチャにゴチャになる… そこで、

=substitute(substitute(A1, "m", ""), " ", "")

↓

= reduce(
  A1,
  {"m", " "},
  lambda(acc, old, substitute(acc, old, ""))
)

function rearrangeData() {
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();  // アクティブなシートを取得
    const data = sheet.getDataRange().getValues();                         // シート内の全データを2次元配列として取得

    let results = [];

    for (let i = 0; i < data.length; i += 2) {              // データを2行ずつ処理
        const code201 = data[i][0];                         // 奇数行にある[201(found)]を取得
        const code300 = data[i + 1] ? data[i + 1][0] : '';  // 偶数行にある[300(extracted)]を取得
        results.push([code201, code300]);                   // [201][300]のペアを配列として保存
    }

    // 結果をシートに書き込む（C列から）
    const startRow = 1; // 書き込み開始行
    const startCol = 3; // 書き込み開始列（C列）
    sheet.getRange(startRow, startCol, results.length, 2).setValues(results); // 書き込み範囲に結果をセット
}

 cf. https://script.google.com/u/0/home/projects/1nQSocTOFaD_rYKe4kZWl9HJGHnQnH1LVxjv8bzsAUzB20EgbD3LiWrbf/edit?pli=1

const PROJECT_ID = 'dev-project-XXXX';
const DATASET_ID = 'datamart';
const TABLE_SUFFIX = '_20240801';

// メニュー追加
function onOpen() {
    let sheet = SpreadsheetApp.getActiveSpreadsheet();
    let entries = [{ name: "Create文生成", functionName: "generateBigQueryCreateStatements" }];
    sheet.addMenu("クエリ生成", entries);
}

// メイン処理
function generateBigQueryCreateStatements() {
    try {
        let spreadsheet = SpreadsheetApp.openById('~youer_sheet_id~')
        let listSheet = spreadsheet.getSheetByName('一覧');
        let data = getSheetData(listSheet);

        // 対象テーブルごとに処理
        let sqlStatements = data
            .filter(row => isTableReady(row))
            .map(row => generateTableCreateStatement(spreadsheet, row));

        // SQL文を別シートに出力
        outputSqlStatements(sqlStatements);

    } catch (error) {
        Logger.log(`エラーが発生しました: ${error.message}`);
    }
}


// 一覧シートのデータを取得
function getSheetData(sheet) {
    let lastRow = sheet.getLastRow();
    return sheet.getRange(2, 1, lastRow - 1, sheet.getLastColumn()).getValues();
}

// 対象テーブルが対応済か確認
function isTableReady(row) {
    let statusColumnIndex = 12; // M列（実装ステータス）
    let tableNameColumnIndex = 6; // G列（テーブル物理名）
    let status = row[statusColumnIndex];
    let tableName = row[tableNameColumnIndex];

    return status === '対応済' && tableName;
}

// 各テーブルのCREATE文を生成
function generateTableCreateStatement(spreadsheet, row) {
    let tableNameColumnIndex = 6;
    let tableName = row[tableNameColumnIndex];

    let schemaSheet = spreadsheet.getSheetByName(tableName);
    if (!schemaSheet) {
        throw new Error(`テーブル ${tableName} に対応するシートが見つかりません。`);
    }

    let schemaData = getSchemaData(schemaSheet);
    let columns = schemaData
        .filter(schemaRow => schemaRow[0] !== '' && schemaRow[0] !== '他、共通カラム')
        .map(schemaRow => formatColumn(schemaRow));

    return `CREATE OR REPLACE TABLE \`${PROJECT_ID}.${DATASET_ID}.${tableName}${TABLE_SUFFIX}\` (\n  ${columns.join(',\n  ')}\n);`;
}

// スキーマシートのデータを取得
function getSchemaData(sheet) {
    let schemaLastRow = sheet.getLastRow();
    return sheet.getRange(2, 2, schemaLastRow - 1, 3).getValues(); // B〜D列のデータ取得
}

// フォーマッター
function formatColumn(schemaRow) {
    let columnName = schemaRow[0];
    let dataType = schemaRow[1];

    if (dataType.toLowerCase() === 'float') {
        dataType = 'FLOAT64';
    }

    let isNullable = schemaRow[2] === '可' || dataType.indexOf('Array') >= 0 ? '' : 'NOT NULL';
    return `\`${columnName}\` ${dataType} ${isNullable}`;
}

// SQL文出力
function outputSqlStatements(sqlStatements) {
    let ss = SpreadsheetApp.getActiveSpreadsheet();
    let outputSheet = ss.getSheetByName('ログ用Create文') || ss.insertSheet('ログ用Create文');

    outputSheet.clear(); // 既存の内容をクリア
    outputSheet.getRange(1, 1, sqlStatements.length, 1).setValues(sqlStatements.map(stmt => [stmt]));
}

// Google カレンダーとシートの設定
const calendar = CalendarApp.getDefaultCalendar();
const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('raw');

// 主関数
function aggregateCalendarEvents() {
    const now = new Date();
    const oneWeekAgo = new Date(now.getTime() - (7 * 24 * 60 * 60 * 1000)); // 1週間前
    const eventTitle = 'work';  // 集計対象のタイトル

    // 1. イベントの取得
    const events = getCalendarEvents(oneWeekAgo, now, eventTitle);

    // 2. 日付ごとの集計
    const dateAggregates = aggregateEventsByDate(events);

    // 3. 集計結果をシートに書き込み
    writeDataToSheet(dateAggregates);
}

// 1. イベントの取得（Google カレンダーAPIを有効化しておくこと）
function getCalendarEvents(startDate, endDate, eventTitle) {
    return calendar.getEvents(startDate, endDate, { search: eventTitle });
}

// 2. 日付ごとの集計
function aggregateEventsByDate(events) {
    let dateAggregates = {};

    events.forEach(function(event) {
        const startTime = event.getStartTime();
        const endTime = event.getEndTime();

        // 日付を取得
        let date = Utilities.formatDate(startTime, Session.getScriptTimeZone(), 'yyyy-MM-dd');
        let durationMillis = endTime - startTime; // ミリ秒単位で差分を計算
        let durationHours = durationMillis / (1000 * 60 * 60); // ミリ秒から時間に変換

        // 日付ごとに合計時間を計算
        if (dateAggregates[date]) {
            dateAggregates[date] += durationHours;
        } else {
            dateAggregates[date] = durationHours;
        }
    });

    return dateAggregates;
}

// 3. 集計結果をシートに書き込み
function writeDataToSheet(dateAggregates) {
    let data = [];

    // 集計結果をデータ配列に変換
    for (let date in dateAggregates) {
        if (dateAggregates.hasOwnProperty(date)) {
            data.push([date, dateAggregates[date]]);
        }
    }

    // シートに転記（既存のデータはクリア）
    if (data.length > 0) {
        sheet.clear();
        // ヘッダー
        sheet.appendRow(['日付', '合計時間']);
        // データを転記
        sheet.getRange(2, 1, data.length, 2).setValues(data);

    } else {
        Logger.log('There were no matching events.');

    }
}

// カレンダーAPIをモックする関数
function mockGetCalendarEvents(startDate, endDate, options) {
    return [
        {
            getStartTime: function() { return new Date('2024-11-01T09:00:00'); },
            getEndTime: function() { return new Date('2024-11-01T10:00:00'); }
        },
        {
            getStartTime: function() { return new Date('2024-11-02T13:00:00'); },
            getEndTime: function() { return new Date('2024-11-02T14:00:00'); }
        }
    ];
}

// テスト用に関数を置き換え
function getCalendarEvents(startDate, endDate, eventTitle) {
    return mockGetCalendarEvents(startDate, endDate, eventTitle);
}

// テスト実行
function testGetCalendarEvents() {
    const events = getCalendarEvents(new Date('2024-10-01'), new Date('2024-11-01'), 'work');
    console.log(events.length);            // 期待される結果: 2件のイベント
    console.log(events[0].getStartTime()); // 期待される結果: 2024-11-01T09:00:00
}
    • GASでは、外部サービスに依存する関数をテストする際に「モック」を使うのが一般的。
    • モックは、外部依存性を仮想的に置き換えて、特定の戻り値を返すようにする手法。
    • e.g. Google カレンダー API をモックするために、CalendarApp.getDefaultCalendar() や calendar.getEvents() を直接呼び出さないようにして、テスト中は仮のデータを使うことができる。
    • i.e. google-apps-script-test などのサードパーティのツールを利用することも可能
